#Start firewall
systemctl enable firewalld
systemctl start firewalld

#Open Port in firewall
firewall-cmd --permanent --add-port=80/tcp
firewall-cmd --reload

#remove port in firewall
firewall-cmd --zone=public --remove-port=80/tcp
firewall-cmd --runtime-to-permanent 
firewall-cmd --reload

#We can check the ports that are opened in the current default zone with ‘--list-ports’.
firewall-cmd --list-ports

#Firewall Status
firewall-cmd --list-all

#Block IP 
firewall-cmd --permanent --add-rich-rule="rule family='ipv4' source address='116.31.116.12' reject"

#Unblock IP
firewall-cmd --remove-rich-rule='rule family=ipv4 source address=192.168.17.112 reject' --permanent
